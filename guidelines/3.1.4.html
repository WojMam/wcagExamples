        return abbrMap.get(context) || abbrMap.get('general') || null;
    }
    
    // Automatycznie dodaj elementy abbr w kontek≈õcie
    processSection(element, context = 'general') {
        const textNodes = this.getTextNodes(element);
        
        this.abbreviations.forEach((contextMap, abbr) => {
            const expansion = this.getExpansion(abbr, context);
            if (!expansion) return;
            
            const regex = new RegExp(`\\b${this.escapeRegex(abbr)}\\b`, 'g');
            
            textNodes.forEach(node => {
                if (regex.test(node.textContent) && !this.processed.has(node)) {
                    this.wrapAbbreviation(node, abbr, expansion);
                    this.processed.add(node);
                }
            });
        });
    }
    
    wrapAbbreviation(textNode, abbr, expansion) {
        const parent = textNode.parentNode;
        const text = textNode.textContent;
        const regex = new RegExp(`\\b(${this.escapeRegex(abbr)})\\b`, 'g');
        
        if (regex.test(text)) {
            const newHTML = text.replace(regex, 
                `<abbr title="${this.escapeHtml(expansion)}">$1</abbr>`
            );
            
            const wrapper = document.createElement('span');
            wrapper.innerHTML = newHTML;
            
            // ZastƒÖp wƒôze≈Ç tekstowy nowym elementem
            parent.insertBefore(wrapper, textNode);
            parent.removeChild(textNode);
        }
    }
    
    getTextNodes(element) {
        const textNodes = [];
        const walker = document.createTreeWalker(
            element,
            NodeFilter.SHOW_TEXT,
            null,
            false
        );
        
        let node;
        while (node = walker.nextNode()) {
            if (node.textContent.trim()) {
                textNodes.push(node);
            }
        }
        
        return textNodes;
    }
    
    escapeRegex(string) {
        return string.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
    }
    
    escapeHtml(string) {
        return string.replace(/[&<>"']/g, (match) => {
            const escapeMap = { 
                '&': '&', '<': '<', '>': '>', 
                '"': '"', "'": ''' 
            };
            return escapeMap[match];
        });
    }
    
    // Generuj s≈Çowniczek dla danego kontekstu
    generateGlossary(context = 'general') {
        const glossary = [];
        
        this.abbreviations.forEach((contextMap, abbr) => {
            const expansion = contextMap.get(context) || contextMap.get('general');
            if (expansion) {
                glossary.push({ abbr, expansion });
            }
        });
        
        return glossary.sort((a, b) => a.abbr.localeCompare(b.abbr));
    }
}

class AutoGlossaryGenerator {
    constructor(options = {}) {
        this.minOccurrences = options.minOccurrences || 2;
        this.commonWords = new Set(options.commonWords || []);
        this.glossaryContainer = options.container || '#abbreviations-glossary';
    }
    
    // Znajd≈∫ wszystkie potencjalne skr√≥ty
    findAbbreviations() {
        const text = document.body.textContent;
        const abbreviations = new Map();
        
        // Wzorce dla r√≥≈ºnych typ√≥w skr√≥t√≥w
        const patterns = [
            /\b[A-Z]{2,6}\b/g,  // Akronimy
            /\b[A-Z][a-z]*\.[A-Za-z.]*/g  // Skr√≥ty z kropkami
        ];
        
        patterns.forEach(pattern => {
            const matches = text.match(pattern) || [];
            matches.forEach(match => {
                if (!this.commonWords.has(match)) {
                    abbreviations.set(match, (abbreviations.get(match) || 0) + 1);
                }
            });
        });
        
        // Filtruj wed≈Çug minimalnej liczby wystƒÖpie≈Ñ
        return new Map([...abbreviations].filter(([abbr, count]) => 
            count >= this.minOccurrences
        ));
    }
    
    // Pr√≥buj znale≈∫ƒá definicje w tek≈õcie
    findDefinitions(abbreviations) {
        const definitions = new Map();
        const bodyText = document.body.textContent;
        
        abbreviations.forEach((count, abbr) => {
            // Szukaj wzorc√≥w: "Full Form (ABBR)" lub "ABBR (Full Form)"
            const patterns = [
                new RegExp(`([^(]+)\\s*\\(\\s*${this.escapeRegex(abbr)}\\s*\\)`, 'gi'),
                new RegExp(`${this.escapeRegex(abbr)}\\s*\\(([^)]+)\\)`, 'gi')
            ];
            
            patterns.forEach(pattern => {
                const matches = bodyText.match(pattern);
                if (matches && matches.length > 0) {
                    const match = matches[0];
                    const fullForm = match.replace(abbr, '').replace(/[()]/g, '').trim();
                    if (fullForm && fullForm !== abbr) {
                        definitions.set(abbr, fullForm);
                    }
                }
            });
        });
        
        return definitions;
    }
    
    // Generuj HTML s≈Çowniczka
    generateGlossaryHTML(definitions) {
        if (definitions.size === 0) return '';
        
        const entries = Array.from(definitions.entries())
            .sort(([a], [b]) => a.localeCompare(b))
            .map(([abbr, definition]) => `
                <dt id="abbr-${abbr.toLowerCase()}"><abbr>${abbr}</abbr></dt>
                <dd>${definition}</dd>
            `).join('');
        
        return `
            <section id="auto-abbreviations-glossary">
                <h2>S≈Çowniczek skr√≥t√≥w</h2>
                <dl>
                    ${entries}
                </dl>
            </section>
        `;
    }
    
    // Dodaj linki do s≈Çowniczka
    linkToGlossary(definitions) {
        definitions.forEach((definition, abbr) => {
            const regex = new RegExp(`\\b${this.escapeRegex(abbr)}\\b`, 'g');
            const elements = document.querySelectorAll('p, span, div');
            
            elements.forEach(element => {
                if (element.innerHTML.includes(abbr)) {
                    element.innerHTML = element.innerHTML.replace(regex, 
                        `<a href="#abbr-${abbr.toLowerCase()}">${abbr}</a>`
                    );
                }
            });
        });
    }
    
    // G≈Ç√≥wna funkcja
    generate() {
        const abbreviations = this.findAbbreviations();
        const definitions = this.findDefinitions(abbreviations);
        
        if (definitions.size > 0) {
            const glossaryHTML = this.generateGlossaryHTML(definitions);
            
            // Dodaj s≈Çowniczek na koniec dokumentu
            const container = document.querySelector(this.glossaryContainer) || 
                             document.querySelector('main') || 
                             document.body;
            
            const glossaryElement = document.createElement('div');
            glossaryElement.innerHTML = glossaryHTML;
            container.appendChild(glossaryElement);
            
            // Dodaj linki
            this.linkToGlossary(definitions);
        }
        
        return definitions;
    }
    
    escapeRegex(string) {
        return string.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
    }
}

// U≈ºycie
const autoGlossary = new AutoGlossaryGenerator({
    minOccurrences: 2,
    commonWords: ['IT', 'OK', 'TV', 'DVD', 'USB'], // Wyklucz powszechne skr√≥ty
    container: 'main'
});

// Automatycznie generuj s≈Çowniczek
document.addEventListener('DOMContentLoaded', () => {
    const definitions = autoGlossary.generate();
    console.log('Wygenerowano definicje dla:', definitions);
});
