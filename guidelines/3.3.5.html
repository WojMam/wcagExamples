class ContextualHelp {
    constructor() {
        this.activeHelp = null;
        this.helpData = new Map();
        this.init();
    }
    
    init() {
        // Inicjalizacja event√≥w
        document.addEventListener('click', this.handleHelpClick.bind(this));
        document.addEventListener('focus', this.handleFieldFocus.bind(this), true);
        document.addEventListener('keydown', this.handleKeyDown.bind(this));
    }
    
    // Rejestracja pomocy dla pola
    registerHelp(fieldId, helpConfig) {
        this.helpData.set(fieldId, {
            content: helpConfig.content,
            trigger: helpConfig.trigger || 'click',
            type: helpConfig.type || 'tooltip',
            position: helpConfig.position || 'below'
        });
    }
    
    // Pokazanie pomocy
    showHelp(fieldId, triggerElement) {
        const helpConfig = this.helpData.get(fieldId);
        if (!helpConfig) return;
        
        // Ukryj aktualnƒÖ pomoc
        this.hideCurrentHelp();
        
        // Utw√≥rz element pomocy
        const helpElement = this.createHelpElement(fieldId, helpConfig);
        
        // Umie≈õƒá pomoc w odpowiednim miejscu
        this.positionHelp(helpElement, triggerElement, helpConfig.position);
        
        // Poka≈º pomoc
        this.displayHelp(helpElement, helpConfig.type);
        
        this.activeHelp = helpElement;
    }
    
    createHelpElement(fieldId, config) {
        const helpElement = document.createElement('div');
        helpElement.className = `contextual-help contextual-help--${config.type}`;
        helpElement.id = `help-${fieldId}-${Date.now()}`;
        helpElement.setAttribute('role', 'tooltip');
        helpElement.setAttribute('aria-live', 'polite');
        
        // Dodaj przycisk zamkniƒôcia
        helpElement.innerHTML = `
            <div class="help-content">
                <button type="button" class="help-close" aria-label="Zamknij pomoc">√ó</button>
                ${config.content}
            </div>
        `;
        
        // Event dla zamkniƒôcia
        helpElement.querySelector('.help-close').addEventListener('click', () => {
            this.hideHelp(helpElement);
        });
        
        return helpElement;
    }
    
    positionHelp(helpElement, triggerElement, position) {
        document.body.appendChild(helpElement);
        
        const triggerRect = triggerElement.getBoundingClientRect();
        const helpRect = helpElement.getBoundingClientRect();
        const viewport = {
            width: window.innerWidth,
            height: window.innerHeight
        };
        
        let top, left;
        
        switch (position) {
            case 'above':
                top = triggerRect.top - helpRect.height - 10;
                left = triggerRect.left;
                break;
            case 'below':
                top = triggerRect.bottom + 10;
                left = triggerRect.left;
                break;
            case 'left':
                top = triggerRect.top;
                left = triggerRect.left - helpRect.width - 10;
                break;
            case 'right':
                top = triggerRect.top;
                left = triggerRect.right + 10;
                break;
            default:
                top = triggerRect.bottom + 10;
                left = triggerRect.left;
        }
        
        // Sprawd≈∫ czy pomoc mie≈õci siƒô w viewport
        if (left + helpRect.width > viewport.width) {
            left = viewport.width - helpRect.width - 20;
        }
        if (left < 20) {
            left = 20;
        }
        if (top + helpRect.height > viewport.height) {
            top = triggerRect.top - helpRect.height - 10;
        }
        if (top < 20) {
            top = 20;
        }
        
        helpElement.style.position = 'fixed';
        helpElement.style.top = `${top}px`;
        helpElement.style.left = `${left}px`;
        helpElement.style.zIndex = '1000';
    }
    
    displayHelp(helpElement, type) {
        switch (type) {
            case 'tooltip':
                helpElement.classList.add('help-tooltip');
                break;
            case 'modal':
                helpElement.classList.add('help-modal');
                this.createOverlay();
                break;
            case 'inline':
                helpElement.classList.add('help-inline');
                break;
        }
        
        helpElement.hidden = false;
        
        // Animacja wej≈õcia
        requestAnimationFrame(() => {
            helpElement.classList.add('help-visible');
        });
        
        // Focus na pierwszym focusable element
        const firstFocusable = helpElement.querySelector('button, a, input, [tabindex]');
        if (firstFocusable) {
            firstFocusable.focus();
        }
    }
    
    hideCurrentHelp() {
        if (this.activeHelp) {
            this.hideHelp(this.activeHelp);
        }
    }
    
    hideHelp(helpElement) {
        if (!helpElement) return;
        
        helpElement.classList.remove('help-visible');
        
        setTimeout(() => {
            if (helpElement.parentNode) {
                helpElement.parentNode.removeChild(helpElement);
            }
            if (this.activeHelp === helpElement) {
                this.activeHelp = null;
            }
        }, 300);
        
        this.removeOverlay();
    }
    
    createOverlay() {
        if (document.querySelector('.help-overlay')) return;
        
        const overlay = document.createElement('div');
        overlay.className = 'help-overlay';
        overlay.addEventListener('click', () => this.hideCurrentHelp());
        document.body.appendChild(overlay);
    }
    
    removeOverlay() {
        const overlay = document.querySelector('.help-overlay');
        if (overlay) {
            overlay.parentNode.removeChild(overlay);
        }
    }
    
    handleHelpClick(event) {
        if (event.target.matches('.help-trigger, .help-trigger *')) {
            event.preventDefault();
            const trigger = event.target.closest('.help-trigger');
            const fieldId = trigger.getAttribute('aria-describedby')?.replace('-help', '');
            if (fieldId) {
                this.showHelp(fieldId, trigger);
            }
        }
    }
    
    handleFieldFocus(event) {
        if (event.target.matches('input, select, textarea')) {
            const fieldId = event.target.id;
            const helpConfig = this.helpData.get(fieldId);
            if (helpConfig && helpConfig.trigger === 'focus') {
                this.showHelp(fieldId, event.target);
            }
        }
    }
    
    handleKeyDown(event) {
        if (event.key === 'Escape' && this.activeHelp) {
            this.hideCurrentHelp();
        }
    }
}
