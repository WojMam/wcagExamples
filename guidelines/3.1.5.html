class TextSimplifier {
    constructor() {
        // S≈Çownik synonim√≥w prostych
        this.synonyms = new Map([
            ['wykorzystaƒá', 'u≈ºyƒá'],
            ['implementowaƒá', 'wdro≈ºyƒá'],
            ['zrealizowaƒá', 'wykonaƒá'],
            ['specyfikacja', 'opis'],
            ['funkcjonalno≈õƒá', 'funkcja'],
            ['procedura', 'spos√≥b'],
            ['metodologia', 'metoda'],
            ['dokumentacja', 'opis'],
            ['administracja', 'zarzƒÖdzanie'],
            ['optymalizacja', 'poprawa']
        ]);
        
        // Wzorce do upraszczania
        this.simplificationRules = [
            {
                pattern: /w celu (.+)/gi,
                replacement: '≈ºeby $1'
            },
            {
                pattern: /jest mo≈ºliwe (.+)/gi,
                replacement: 'mo≈ºna $1'
            },
            {
                pattern: /nale≈ºy (.+)/gi,
                replacement: 'musisz $1'
            },
            {
                pattern: /w przypadku gdy (.+)/gi,
                replacement: 'je≈õli $1'
            }
        ];
    }
    
    simplifyText(text) {
        let simplified = text;
        
        // ZastƒÖp skomplikowane s≈Çowa prostszymi
        for (let [complex, simple] of this.synonyms) {
            const regex = new RegExp('\\b' + complex + '\\b', 'gi');
            simplified = simplified.replace(regex, simple);
        }
        
        // Zastosuj wzorce upraszczania
        for (let rule of this.simplificationRules) {
            simplified = simplified.replace(rule.pattern, rule.replacement);
        }
        
        // Podziel d≈Çugie zdania
        simplified = this.splitLongSentences(simplified);
        
        // Zmie≈Ñ biernƒÖ stronƒô na czynnƒÖ
        simplified = this.convertPassiveToActive(simplified);
        
        return simplified;
    }
    
    splitLongSentences(text) {
        const sentences = text.split(/(?<=[.!?])\s+/);
        const simplified = [];
        
        for (let sentence of sentences) {
            const words = sentence.split(/\s+/);
            if (words.length > 20) {
                // Znajd≈∫ punkt podzia≈Çu (sp√≥jniki)
                const conjunctions = ['i', 'ale', 'oraz', 'albo', 'jednak', 'poniewa≈º', 'dlatego'];
                let splitPoint = -1;
                
                for (let i = 10; i < words.length - 5; i++) {
                    if (conjunctions.includes(words[i].toLowerCase())) {
                        splitPoint = i;
                        break;
                    }
                }
                
                if (splitPoint > -1) {
                    const firstPart = words.slice(0, splitPoint).join(' ') + '.';
                    const secondPart = words.slice(splitPoint + 1).join(' ');
                    simplified.push(firstPart);
                    simplified.push(secondPart);
                } else {
                    simplified.push(sentence);
                }
            } else {
                simplified.push(sentence);
            }
        }
        
        return simplified.join(' ');
    }
    
    convertPassiveToActive(text) {
        // Podstawowe wzorce strony biernej w polskim
        const passivePatterns = [
            {
                pattern: /(.+) zosta≈Ç (.+) przez (.+)/gi,
                replacement: '$3 $2 $1'
            },
            {
                pattern: /(.+) zostanie (.+)/gi,
                replacement: 'kto≈õ $2 $1'
            }
        ];
        
        let converted = text;
        for (let pattern of passivePatterns) {
            converted = converted.replace(pattern.pattern, pattern.replacement);
        }
        
        return converted;
    }
    
    generateSimplifiedVersion(originalText, targetLevel = 50) {
        let simplified = this.simplifyText(originalText);
        let attempts = 0;
        const maxAttempts = 3;
        
        while (attempts < maxAttempts) {
            const score = this.calculateReadabilityScore(simplified);
            if (score >= targetLevel) break;
            
            // Dodatkowe uproszczenia
            simplified = this.furtherSimplify(simplified);
            attempts++;
        }
        
        return {
            text: simplified,
            score: this.calculateReadabilityScore(simplified),
            improvements: this.getImprovements(originalText, simplified)
        };
    }
    
    furtherSimplify(text) {
        // Zamie≈Ñ przecinki na kropki gdzie to mo≈ºliwe
        let simplified = text.replace(/([^,]{20,}), ([^,]{10,})/g, '$1. $2');
        
        // Usu≈Ñ niepotrzebne s≈Çowa
        const fillerWords = ['oczywi≈õcie', 'jednak≈ºe', 'w rzeczywisto≈õci', 'rzeczywi≈õcie'];
        for (let word of fillerWords) {
            const regex = new RegExp('\\b' + word + '\\b,?\\s*', 'gi');
            simplified = simplified.replace(regex, '');
        }
        
        return simplified;
    }
    
    calculateReadabilityScore(text) {
        // Uproszczona implementacja Flesch
        const sentences = text.split(/[.!?]+/).filter(s => s.trim());
        const words = text.split(/\s+/).filter(w => w.trim());
        
        if (sentences.length === 0 || words.length === 0) return 0;
        
        const avgWordsPerSentence = words.length / sentences.length;
        const avgSyllablesPerWord = words.reduce((total, word) => 
            total + this.countSyllables(word), 0) / words.length;
        
        return 206.835 - (1.015 * avgWordsPerSentence) - (84.6 * avgSyllablesPerWord);
    }
    
    countSyllables(word) {
        const vowels = 'aeiouyƒÖƒô√≥';
        let count = 0;
        let prevWasVowel = false;
        
        for (let char of word.toLowerCase()) {
            const isVowel = vowels.includes(char);
            if (isVowel && !prevWasVowel) count++;
            prevWasVowel = isVowel;
        }
        
        return Math.max(1, count);
    }
    
    getImprovements(original, simplified) {
        const originalScore = this.calculateReadabilityScore(original);
        const simplifiedScore = this.calculateReadabilityScore(simplified);
        
        return {
            scoreImprovement: simplifiedScore - originalScore,
            lengthReduction: ((original.length - simplified.length) / original.length * 100).toFixed(1) + '%',
            readabilityGain: simplifiedScore >= 50 ? 'OsiƒÖgniƒôto cel WCAG' : 'Wymaga dalszej pracy'
        };
    }

class MultilingualReadabilityManager {
    constructor() {
        this.languages = {
            'pl': {
                name: 'Polski',
                vowels: 'aeiouyƒÖƒô√≥',
                avgSyllablesPerWord: 2.1,
                avgWordsPerSentence: 16,
                complexWordThreshold: 3
            },
            'en': {
                name: 'English',
                vowels: 'aeiouy',
                avgSyllablesPerWord: 1.5,
                avgWordsPerSentence: 15,
                complexWordThreshold: 3
            }
        };
        
        this.readabilityFormulas = {
            'flesch': this.calculateFlesch,
            'kincaid': this.calculateKincaid,
            'gunning': this.calculateGunningFog
        };
    }
    
    analyzeMultilingualContent(content, primaryLanguage = 'pl') {
        const sections = this.detectLanguageSections(content);
        const results = {};
        
        for (let [lang, text] of sections) {
            results[lang] = this.analyzeLanguageSection(text, lang);
        }
        
        return {
            primary: primaryLanguage,
            sections: results,
            overallCompliance: this.checkOverallCompliance(results),
            recommendations: this.getMultilingualRecommendations(results)
        };
    }
    
    detectLanguageSections(content) {
        // Uproszczone wykrywanie sekcji jƒôzykowych po atrybucie lang
        const sections = new Map();
        const parser = new DOMParser();
        const doc = parser.parseFromString(content, 'text/html');
        
        // G≈Ç√≥wny jƒôzyk
        const mainLang = doc.documentElement.lang || 'pl';
        let mainText = doc.body.textContent;
        
        // Znajd≈∫ elementy z innym jƒôzykiem
        const foreignElements = doc.querySelectorAll('[lang]:not([lang="' + mainLang + '"])');
        
        foreignElements.forEach(element => {
            const lang = element.getAttribute('lang');
            const text = element.textContent;
            
            if (!sections.has(lang)) {
                sections.set(lang, '');
            }
            sections.set(lang, sections.get(lang) + ' ' + text);
            
            // Usu≈Ñ z tekstu g≈Ç√≥wnego
            mainText = mainText.replace(text, '');
        });
        
        sections.set(mainLang, mainText);
        
        return sections;
    }
    
    analyzeLanguageSection(text, language) {
        const langConfig = this.languages[language] || this.languages['pl'];
        const analysis = this.basicTextAnalysis(text, langConfig);
        
        return {
            language: langConfig.name,
            wordCount: analysis.words,
            sentenceCount: analysis.sentences,
            avgWordsPerSentence: analysis.avgWordsPerSentence,
            readabilityScores: {
                flesch: this.calculateFlesch(analysis, langConfig),
                kincaid: this.calculateKincaid(analysis, langConfig),
                gunning: this.calculateGunningFog(analysis, langConfig)
            },
            wcagCompliant: this.checkWCAGCompliance(analysis, langConfig),
            suggestions: this.getLanguageSpecificSuggestions(analysis, langConfig)
        };
    }
    
    basicTextAnalysis(text, langConfig) {
        const sentences = text.split(/[.!?]+/).filter(s => s.trim().length > 0);
        const words = text.split(/\s+/).filter(w => w.trim().length > 0);
        const syllables = words.reduce((total, word) => 
            total + this.countSyllables(word, langConfig.vowels), 0);
        
        return {
            sentences: sentences.length,
            words: words.length,
            syllables: syllables,
            avgWordsPerSentence: words.length / sentences.length,
            avgSyllablesPerWord: syllables / words.length,
            complexWords: this.getComplexWords(words, langConfig)
        };
    }
    
    countSyllables(word, vowels) {
        let count = 0;
        let prevWasVowel = false;
        
        for (let char of word.toLowerCase()) {
            const isVowel = vowels.includes(char);
            if (isVowel && !prevWasVowel) count++;
            prevWasVowel = isVowel;
        }
        
        return Math.max(1, count);
    }
    
    getComplexWords(words, langConfig) {
        return words.filter(word => 
            this.countSyllables(word, langConfig.vowels) >= langConfig.complexWordThreshold
        );
    }
    
    calculateFlesch(analysis, langConfig) {
        return 206.835 - (1.015 * analysis.avgWordsPerSentence) - (84.6 * analysis.avgSyllablesPerWord);
    }
    
    calculateKincaid(analysis, langConfig) {
        return (0.39 * analysis.avgWordsPerSentence) + (11.8 * analysis.avgSyllablesPerWord) - 15.59;
    }
    
    calculateGunningFog(analysis, langConfig) {
        const complexWordPercentage = (analysis.complexWords.length / analysis.words) * 100;
        return 0.4 * (analysis.avgWordsPerSentence + complexWordPercentage);
    }

        const fleschScore = this.calculateFlesch(analysis, langConfig);
        const gradeLevel = this.calculateKincaid(analysis, langConfig);
        
        return {
            flesch: fleschScore >= 50,
            grade: gradeLevel <= 12,
            overall: fleschScore >= 50 && gradeLevel <= 12
        };
    }
    
    checkOverallCompliance(results) {
        let compliantSections = 0;
        let totalSections = 0;
        
        for (let [lang, analysis] of Object.entries(results)) {
            totalSections++;
            if (analysis.wcagCompliant.overall) {
                compliantSections++;
            }
        }
        
        return {
            compliantSections: compliantSections,
            totalSections: totalSections,
            percentage: (compliantSections / totalSections * 100).toFixed(1),
            isCompliant: compliantSections === totalSections
        };
    }
    
    getMultilingualRecommendations(results) {
        const recommendations = [];
        
        for (let [lang, analysis] of Object.entries(results)) {
